## 💡 Usage

### Basic Usage

```python
from ai_lab_super import DataHub, MLLab, DLAStudio

# Initialize components
data_hub = DataHub()
ml_lab = MLLab()
dl_studio = DLAStudio()

# Load and process data
data = data_hub.load_data('my_dataset.csv')
data_processed = data_hub.preprocess(data)

# Train ML model
model = ml_lab.train(
    data=data_processed,
    target='target_column',
    model_type='auto',  # AutoML
    hyperparameter_tuning=True
)

# Evaluate model
metrics = ml_lab.evaluate(model, data_processed)
print(metrics)

# Make predictions
predictions = model.predict(new_data)
```

```python
from ai_lab_super import HyperparameterTuner

# Define parameter grid
param_grid = {
    'n_estimators': [50, 100, 200, 500],
    'max_depth': [5, 10, 20, None],
    'learning_rate': [0.01, 0.05, 0.1, 0.2],

# Initialize tuner
tuner = HyperparameterTuner(

    method='bayesian',  # or 'grid', 'random'
    cv_folds=5,
    scoring='accuracy',
    n_trials=100,
    time_budget=3600  # 1 hour
)

# Run optimization
best_params = tuner.optimize(X_train, y_train)
```
